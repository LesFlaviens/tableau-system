async function login() { const tenantID = document.getElementById('tenantID').value.trim(); const btn = document.getElementById('login-btn'); const msg = document.getElementById('error-msg'); if (!tenantID) return; btn.innerText = "Vérification des licences..."; btn.disabled = true; msg.style.display = 'none'; // 1. On fabrique une empreinte pour cette tablette (ou on récupère celle existante) let deviceID = localStorage.getItem('ichef_device_id'); if (!deviceID) { deviceID = 'device_' + Math.random().toString(36).substr(2, 9) + '_' + Date.now(); localStorage.setItem('ichef_device_id', deviceID); } try { // 2. On envoie l'ID du restaurant ET l'empreinte de la tablette au serveur const response = await fetch('https://tableau-system.onrender.com/api/register-device', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ tenantID: tenantID, deviceID: deviceID }) }); const result = await response.json(); if (result.success) { // Autorisation accordée localStorage.setItem('ichef_tenant_id', tenantID); localStorage.setItem('ichef_client_name', result.clientName); btn.style.background = "#4ade80"; btn.style.color = "#000"; btn.innerText = "Licence Validée. Redirection..."; setTimeout(() => { window.location.href = '/logiciel.html'; }, 1000); } else { // Blocage (Suspendu ou trop d'écrans) msg.innerText = result.error || "Accès refusé."; msg.style.display = 'block'; btn.innerText = "ACCÉDER AU SYSTÈME"; btn.disabled = false; } } catch (error) { msg.innerText = "Erreur de connexion au serveur."; msg.style.display = 'block'; btn.innerText = "ACCÉDER AU SYSTÈME"; btn.disabled = false; } }
