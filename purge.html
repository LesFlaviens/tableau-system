unction manage(type, action, idx = null) { if(action === 'add') { layout.push({ id: Date.now(), type, label: type === 'table' ? "T" + (layout.filter(i=>i.type==='table').length + 1) : "ZONE", l:150, t:150, w:100, h:80 }); } if(action === 'delete' && idx !== null) { layout.splice(idx, 1); } if(action === 'rename' && idx !== null) { const name = prompt("Nom :", layout[idx].label); if(name) layout[idx].label = name; } if(action === 'move' && idx !== null) { layout[idx].l += 10; layout[idx].t += 10; } if(action === 'resize' && idx !== null) { layout[idx].w += 20; layout[idx].h += 20; } save(); render(); }
